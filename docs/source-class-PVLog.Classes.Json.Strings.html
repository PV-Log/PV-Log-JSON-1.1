class Strings extends PowerSensor
{

    /**
     * Add the string to an inverter
     *
     * @param  Inverter $inverter
     * @return self For fluid interface
     */
    public function addToInverter(Inverter $inverter)
    {
        $inverter->addString($this);
        return $this;
    }

}
